import sys
#3번과 같은 유형

def DFS(depth):
    if depth == M:
        for i in range(M):
            print(ans[i],end=' ')
        print()
        return
    for i in range(N):
        ans[depth]=lst[i]
        DFS(depth+1)

N, M = map(int, sys.stdin.readline().split())
lst = list(map(int, sys.stdin.readline().split()))
lst.sort()
ans = [0 for _ in range(M)]
DFS(0)
