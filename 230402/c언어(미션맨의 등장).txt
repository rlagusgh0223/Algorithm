#include<stdio.h>
int main(){
    int a = 1;
    int b = 2;
    int c = 3;
    
    int *p;
    // 첫번째 미션 : 아파트의 각 집에 방문하여 문에 적힌 암호 확인
    p = &a;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p);  // 미션맨이 방문하는 곳 주소 : -2073693324, 암호 : 1
    p = &b;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p);  // 미션맨이 방문하는 곳 주소 : -2073693320, 암호 : 2
    p = &c;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p);  // 미션맨이 방문하는 곳 주소 : -2073693316, 암호 : 3
    
    // 두번째 미션 : 각 암호에 3을 곱해라
    p = &a;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p*3);  // 미션맨이 방문하는 곳 주소 : -2073693324, 암호 : 3
    p = &b;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p*3);  // 미션맨이 방문하는 곳 주소 : -2073693320, 암호 : 6
    p = &c;
    printf("미션맨이 방문하는 곳 주소 : %d, 암호 : %d\n", p, *p*3);  // 미션맨이 방문하는 곳 주소 : -2073693316, 암호 : 9
    return 0;
}
