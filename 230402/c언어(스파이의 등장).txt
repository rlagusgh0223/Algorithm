// 스파이 : 미션맨이 바꾼 암호에서 2를 빼라
#include<stdio.h>
int main(){
    int a = 5;
    printf("a의 주소 : %d, 암호 : %d\n", &a, a);
    
    int *p;
    p = &a;
    *p = *p * 3;
    printf("미션맨이 암호 바꾼 주소 : %d, 암호 : %d\n", &a, a);
    int *s = p;
    s = &a;
    *s = *s - 2;
    printf("스파이가 조작한 주소 : %d, 암호 : %d\n", s, *s);
    
    // a의 주소
    printf("a의 주소 : %d, 암호 : %d\n", &a, a);
    
    // 포인터 변수 자체의 주소
    
    printf("미션맨의 주소 : %d\n", &p);
    printf("스파이의 주소 : %d", &s);
    return 0;
}

// a의 주소 : -800643564, 암호 : 5
// 미션맨이 암호 바꾼 주소 : -800643564, 암호 : 15
// 스파이가 조작한 주소 : -800643564, 암호 : 13
// a의 주소 : -800643564, 암호 : 13
// 미션맨의 주소 : -800643560
// 스파이의 주소 : -800643552
