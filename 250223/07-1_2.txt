#include<iostream>
#include<cstring>
using namespace std;

class MyFriendInfo {
private:
	char* name;
	int age;
public:
	MyFriendInfo(const char* name, int age) {
		this->name = new char[strlen(name)+1];
		strcpy(this->name, name);
		this->age = age;
	}
	void ShowMyFriendInfo() {
		cout << "이름 : " << name << endl;
		cout << "나이 : " << age << endl;
	}
	~MyFriendInfo() {
		delete[] name;
	}
};

class MyFriendDetailInfo :public MyFriendInfo {
private:
	char* addr;
	char* phone;
public:
	MyFriendDetailInfo(const char* name, int age, const char* addr, const char* phone) :MyFriendInfo(name, age) {
		this->addr = new char[strlen(addr)+1];
		this->phone = new char[strlen(phone)+1];
		strcpy(this->addr, addr);
		strcpy(this->phone, phone);
	}
	void ShowMyFirendDetailInfo() {
		ShowMyFriendInfo();
		cout << "주소 : " << addr << endl;
		cout << "전화 : " << phone << endl;
	}
	~MyFriendDetailInfo() {
		delete[] addr;
		delete[] phone;
	}
};

int main() {
	MyFriendDetailInfo f1("가", 17, "나", "다");
	f1.ShowMyFirendDetailInfo();
	return 0;
}
