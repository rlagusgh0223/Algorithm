import sys
N, M = map(int,sys.stdin.readline().split())

if N == 1:  #세로가 하나일때 위로 갈 수 없으므로 답은 1
    print('1')
elif N == 2:
    #세로가 2이면 세로로 한칸 옮기는 방법밖에 안되고,
    #4번을 넘어가면 다른 이동방법도 한번은 써야된다는 제약이 있어서 4번을 넘길 수 없다
    print(min(4,(M+1)//2))
elif N >= 3:
    if M >= 7:
    #7은 4번을 다 사용했을때 갈 수 있는 거리이다.
    #7번 중 두칸은 비게되므로(가로로 두칸 가는 방법 두개) 가로-2
        print(M-2)
    elif M < 7:
    #너비가 7보다 작다면 어차피 4번 못쓰니까 최대한 많이 가는거 찾는
        print(min(4,M))
