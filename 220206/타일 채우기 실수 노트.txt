d[0] = 1

d[2] = d[2-2]*3 = d[0]*3 = 1*3 = 3

d[4] = d[4-2]*3 + 2*d[4-4] = d[2]*3 + 2*d[0] = 3*3 + 2*1 = 9 + 2 = 11

d[6] = d[6-2]*3 + 2*d[6-4] + 2*d[6-4-2] = d[4]*3 + 2*d[2] + 2*d[0] = 11*3 + 2*3 + 2*1 = 33 + 6 + 2 = 41

d[8] = d[8-2]*3 + 2*d[8-4] + 2*d[8-4-2] + 2*d[8-4-2-2] = d[6]*3 + 2*d[4] + 2*d[2] + 2*d[0] = 123 + 22 + 6 + 2 = 153
